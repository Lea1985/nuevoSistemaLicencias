import Sequelize from 'sequelize';
import db from '../config/db.js';

export const Articulos = db.define('articulo', {
    idArticulo: {
        type: Sequelize.INTEGER,
        primaryKey: true,
        autoIncrement: true
    },
    articulo: {
        type: Sequelize.STRING,
        allowNull: false // Indica que este campo no puede ser nulo
    },
    sueldo: {
        type: Sequelize.STRING
    },
    ingreso: {
        type: Sequelize.DATE // Cambiado a tipo DATE para almacenar fechas
    },
    requisitos: {
        type: Sequelize.TEXT // Cambiado a tipo TEXT para almacenar texto largo
    },
    cantDias: {
        type: Sequelize.STRING
    },
    dias: {
        type: Sequelize.STRING
    },
    generaReemplazo: {
        type: Sequelize.STRING
    }
}, {
    tableName: 'articulo' // Especifica el nombre de la tabla
});
